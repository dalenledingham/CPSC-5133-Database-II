CREATE TABLE PersonTypes
(
    typeID INTEGER,
    typeInitial VARCHAR,
    typeName VARCHAR,
    PRIMARY KEY (typeID AUTOINCREMENT)
);
CREATE TABLE Persons
(
    pID INTEGER,
    personType INTEGER,
    firstName VARCHAR,
    lastName VARCHAR UNIQUE,
);
CREATE TABLE Diagnoses
(
    dID INTEGER,
    diagnosisName VARCHAR UNIQUE,
    PRIMARY KEY (dID AUTOINCREMENT)
);
CREATE TABLE EmergencyContacts
(
    ecID INTEGER,
    ecName VARCHAR,
    ecPhone VARCHAR,
    PRIMARY KEY (ecID AUTOINCREMENT)
);
CREATE TABLE Treatments
(
    tID INTEGER,
    treatmentName VARCHAR UNIQUE,
    treatmentType VARCHAR,
    PRIMARY KEY (tID AUTOINCREMENT)
);
CREATE TABLE PatientTreatments
(
    ptID INTEGER,
    patient INTEGER,
    treatment INTEGER,
    orderingDoctor INTEGER,
    administerTime VARCHAR,
    PRIMARY KEY (ptID AUTOINCREMENT),
    FOREIGN KEY (patient) REFERENCES Persons (pID),
    FOREIGN KEY (treatment) REFERENCES Treatments (tID),
    FOREIGN KEY (orderingDoctor) REFERENCES Persons (pID)
);
CREATE TABLE Admissions
(
    aID INTEGER,
    patientID INTEGER,
    firstName VARCHAR,
    lastName VARCHAR,
    initialDiagnosis INTEGER,
    primaryDoctor INTEGER,
    insurancePolicy VARCHAR,
    insuranceCompany VARCHAR,
    emergencyContact INTEGER,
    admitDate VARCHAR,
    dischargeDate VARCHAR,
    PRIMARY KEY (aID AUTOINCREMENT),
    FOREIGN KEY (patientID) REFERENCES Persons (pID),
    FOREIGN KEY (initialDiagnosis) REFERENCES Diagnoses (dID),
    FOREIGN KEY (primaryDoctor) REFERENCES Persons (pID),
    FOREIGN KEY (emergencyContact) REFERENCES EmergencyContacts (eID)
);
CREATE TABLE Rooms
(
    rID INTEGER,
    roomNum INTEGER UNIQUE,
    isOccupied VARCHAR,
    patient INTEGER,
    admitDate VARCHAR,
    PRIMARY KEY (rID AUTOINCREMENT),
    FOREIGN KEY (patient) REFERENCES Persons (pID)
);
CREATE TABLE AdditionalDoctors
(
    adID INTEGER,
    patient INTEGER,
    doctor INTEGER,
    PRIMARY KEY (adID AUTOINCREMENT),
    FOREIGN KEY (patient) REFERENCES Persons (pID),
    FOREIGN KEY (doctor) REFERENCES Persons (pID)
);
