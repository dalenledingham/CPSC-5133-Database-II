CREATE TABLE Rooms
(
    roomID INTEGER PRIMARY KEY,
    patientID VARCHAR(10),
    admitDate DATETIME, (?)
    employeeID VARCHAR(10),
    FOREIGN KEY (patientID) REFERENCES Patients (patientID),
    FOREIGN KEY (employeeID) REFERENCES Employees (employeeID)
);

CREATE TABLE Patients
(
    patientID VARCHAR(10) PRIMARY KEY,
    pFirstName VARCHAR(20),
    pLastname VARCHAR(20) UNIQUE,
    employeeID INTEGER,
    emergencyID INTEGER,
    insurancePolicy VARCHAR(20),
    insuranceCompany VARCHAR(20),
    admitDate DATETIME,
    dischargeDate DATETIME,
    FOREIGN KEY (employeeID) REFERENCES Employees (employeeID),
    FOREIGN KEY (emergencyID) REFERENCES EmergencyContacts (emergencyID),
);

CREATE TABLE Treatments
(
    treatmentID VARCHAR(10) PRIMARY KEY,
    treatmentType CHAR(1), // P for procedure or M for medication
    treatmentName VARCHAR(20),
);

CREATE TABLE OrderedTreatments
(
    treatmentID VARCHAR(10),
    patientID VARCHAR(10),
    employeeID VARCHAR(10),
    orderTimestamp DATETIME,
    FOREIGN KEY (treatmentID) REFERENCES Treatments (treatmentID),
    FOREIGN KEY (patientID) REFERENCES Patients (patientID),
    FOREIGN KEY (employeeID) REFERENCES Employees (employeeID)
);

CREATE TABLE AdministeredTreatments
(
    treatmentID VARCHAR(10),
    patientID VARCHAR(10),
    employeeID VARCHAR(10),
    administerTimestamp DATETIME,
    FOREIGN KEY (treatmentID) REFERENCES Treatments (treatmentID),
    FOREIGN KEY (patientID) REFERENCES Patients (patientID),
    FOREIGN KEY (employeeID) REFERENCES Employees (employeeID)
);

CREATE TABLE Diagnoses
(
    diagnosisID VARCHAR(10) PRIMARY KEY,
    diagnosisName VARCHAR(20)
);

CREATE TABLE PatientDiagnoses
(
    diagnosisID VARCHAR(10),
    patientID VARCHAR(10),
    admitDate DATETIME,
    FOREIGN KEY (diagnosisID) REFERENCES Diagnoses (diagnosisID),
    FOREIGN KEY (patientID) REFERENCES Patients (patientID),
    FOREIGN KEY (admitDate) REFERENCES Patients (admitDate)
);

CREATE TABLE Employees
(
    employeeID VARCHAR(10) PRIMARY KEY,
    eFirstName VARCHAR(20),
    eLastName VARCHAR(20) UNIQUE,
    jobCategory CHAR(1) // D for doctor, A for administrator, N for nurst, or T for technician
);

CREATE TABLE EmergencyContacts
(
    emergencyID VARCHAR(10) PRIMARY KEY,
    ecFirstName VARCHAR(20),
    ecLastName VARCHAR(20) UNIQUE,
    phone VARCHAR(12)
);




Room Information
-	Number
-	Occupied/Unoccupied
-	Patient Name
-	Patient Admission Date
-	Assignment Administrator

Patient Information
-	ID Number PRIMARY KEY
-	First Name
-	Last Name UNIQUE
-	Primary/Admitting Doctor
-	Additional Doctors (can have multiple values)
-	Diagnosis (can have multiple values, linked to Admission Date)
-	Emergency Contact
-	Insurance Policy
-	Insurance Company
-	Admission Date (can have multiple values)
-	Discharge Date
-	Discharge Administrator

Treatment Information
-	ID Number PRIMARY KEY
-	Patient Name
-	Treatment Type (P or M)
-	Treatment Name
-	Ordering Doctor
-	Order Timestamp
-	Administrator (who gave the treatment to the patient)
-	Admission Timestamp

Diagnosis Information
-	ID Number PRIMARY KEY
-	Diagnosis Name

Employee Information
-	ID Number PRIMARY KEY
-	First Name
-	Last Name UNIQUE
-	Job Category (D, A, N, or T)

Emergency Contact Information
-	ID Number PRIMARY KEY
-	First Name
-	Last Name UNIQUE
-	Phone
