DROP TABLE Rooms;
DROP TABLE Patients;
DROP TABLE Treatments;
DROP TABLE OrderedTreatments;
DROP TABLE AdministeredTreatments;
DROP TABLE Diagnoses;
DROP TABLE Employees;
DROP TABLE EmergencyContacts;
DROP TABLE AdditionalDoctors;

CREATE TABLE Rooms
(
    roomNum INTEGER,
    pFirstName VARCHAR,
    pLastName VARCHAR UNIQUE,
    admitDate VARCHAR,
    PRIMARY KEY (roomNum)
);
CREATE TABLE Patients
(
    pid INTEGER,
    pFirstName VARCHAR,
    pLastName VARCHAR UNIQUE,
    roomNum INTEGER,
    ecName VARCHAR,
    ecPhone VARCHAR,
    insPolicy VARCHAR,
    insCompany VARCHAR,
    eLastName VARCHAR,
    diagnosisName VARCHAR,
    admitDate VARCHAR,
    dischargeDATE VARCHAR,
    PRIMARY KEY (pid AUTOINCREMENT)
);
CREATE TABLE Treatments
(
    treatmentType VARCHAR,
    treatmentName VARCHAR,
    PRIMARY KEY (treatmentType, treatmentName)
);
CREATE TABLE OrderedTreatments
(
    treatmentName VARCHAR,
    pLastName VARCHAR,
    eLastName VARCHAR,
    orderTime VARCHAR,
    PRIMARY KEY (pLastName, treatmentName, orderTime),
    FOREIGN KEY (treatmentName) REFERENCES Treatments (treatmentName)
);
CREATE TABLE AdministeredTreatments
(
    treatmentName VARCHAR,
    pLastName VARCHAR,
    eLastName VARCHAR,
    administerTime VARCHAR,
    PRIMARY KEY (pLastName, treatmentName, administerTime),
    FOREIGN KEY (treatmentName) REFERENCES Treatments (treatmentName)
);
CREATE TABLE Diagnoses
(
    diagnosisName VARCHAR,
    pFirstName VARCHAR,
    pLastName VARCHAR,
    admitDate VARCHAR,
    dischargeDate VARCHAR,
    PRIMARY KEY (diagnosisName, pLastName, admitDate)
);
CREATE TABLE Employees
(
    eid INTEGER,
    eType VARCHAR,
    eFirstName VARCHAR,
    eLastName VARCHAR UNIQUE,
    PRIMARY KEY (eid AUTOINCREMENT)
);
CREATE TABLE EmergencyContacts
(
    pFirstName VARCHAR,
    pLastName VARCHAR,
    ecName VARCHAR,
    ecPhone VARCHAR,
    PRIMARY KEY (pLastName, ecName)
);
CREATE TABLE AdditionalDoctors
(
    eLastName VARCHAR,
    pLastName VARCHAR,
    PRIMARY KEY (eLastName, pLastName)
);
